最长无重复字符子串

使用滑动窗口的方法，查找以每个字符开头最长的无重复子串


//max lengh of substr without repeat charactor
package main

import "fmt"

//return maxlen and the indexs of byte in the str
func FindMaxSubstr(str string) (int, []int) {
    strmap := make(map[byte]int, 0)
    strres := make([]byte, 0)
    reslen := 0
    result := make([]int, 0)
    i := 0
    j := 0
    n := len(str)
    for i < n && j < n {
        fmt.Println(i, j, strmap, strres, reslen)
        //使用map记录每次遍历过程中无重复字符以及所在的位置信息
        if _, ok := strmap[str[j]]; !ok {
            strmap[str[j]] = j
            j++
        } else {
            //遇到重复字符，开始处理：
            //与strres比较长短，以及清空初始化操作
            //将i设置到j对应的字符第一次出现的下个位置，进行下一个过程
            reslen = len(strres)
            resstrij := len(strmap)
            if reslen < resstrij {
                strres = []byte(str[i:j])
                result = make([]int, 0)
                result = append(result, i)
            }
            if reslen == resstrij {
                result = append(result, i)
            }
            i = strmap[str[j]] + 1
            j = i
            strmap = make(map[byte]int, 0)
        }
    }
    return reslen, result
}

func main() {
    str := "abccdeace"
    maxlen, idxs := FindMaxSubstr(str)
    for _, idx := range idxs {
        fmt.Printf("the max sucstr of %v is %v\n", str, str[idx:idx+maxlen])
    }
}

无符号整数1的数量

无符号整数1的数量

  1. 朴素法
    与无符号整数的位数有关，与1按位&，然后整数右移一位

   func numberOf1(num uint64) int {
    c := 0
    for num > 0 {
        bn := num & 1
        if bn == 1 {
            c++
        }
        num = num >> 1
    }
    return c
}
  1. 快速法只与无符号整数中1的个数有关，num =（num-1)+1，num & (num -1)即可去掉num中最右侧的1
   func numberOf12(num uint64) int {
    c := 0
    for num > 0 {
        num = num & (num - 1)
        c++
    }
    return c
}
